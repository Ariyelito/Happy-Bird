            self.pos_y += self.gravite
        if self.saut:
            self.pos_y -= 4
            self.saut_temps += 1
            if self.saut_temps > 40:
                self.saut = False
                self.saut_temps = 0

    def detect_collision(self, rect):
        if self.rect.colliderect(rect):
            return True
        return False
